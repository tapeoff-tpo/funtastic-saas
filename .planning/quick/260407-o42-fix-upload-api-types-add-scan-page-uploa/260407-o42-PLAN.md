2. **Add `handleUpload` function:**
   ```typescript
   const handleUpload = useCallback(async () => {
     if (uploading) return
     setUploading(true)
     setUploadResult(null)
     try {
       const res = await fetch('/api/shipping/upload', {
         method: 'POST',
         headers: { 'Content-Type': 'application/json' },
       })
       const data = await res.json()
       setUploadResult({ uploaded: data.uploaded, failed: data.failed, total: data.total })
       // TTS feedback
       if (data.uploaded > 0 || data.failed > 0) {
         speak(`${data.uploaded}건 전송 완료. ${data.failed > 0 ? `${data.failed}건 실패.` : ''}`)
       } else {
         speak('전송할 건이 없습니다')
       }
     } catch {
       speak('전송 중 오류가 발생했습니다')
       setUploadResult(null)
     } finally {
       setUploading(false)
       refocus()
     }
   }, [uploading, refocus])
   ```
