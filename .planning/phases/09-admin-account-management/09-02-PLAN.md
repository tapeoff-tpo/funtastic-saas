Every mutation is a Server Action with a `super_admin` role check (or owner-self check for self-password-change), uses Drizzle transactions for atomicity (profile change + audit row in one commit), and uses `SELECT ...    import 'server-only'
   ```ts
   import { createClient } from '@/lib/supabase/server'
   import { db } from '@/lib/db'
   import { userProfiles, type UserProfile } from '@/lib/db/schema'
   import { eq } from 'drizzle-orm'
   import type { User } from '@supabase/supabase-js'

   export const BAN_FOREVER = '876000h' as const

   export function getInitialPassword(): string {
     const pw = process.env.INITIAL_USER_PASSWORD
     if (!pw || pw.length < 8) {
       throw new Error('INITIAL_USER_PASSWORD 환경변수가 누락되었거나 8자 미만입니다.')
     }
     return pw
   }

   export type ActorContext = { user: User; profile: UserProfile }

   export async function assertSuperAdmin(): Promise<ActorContext> {
     const supabase = await createClient()
     const { data: { user } } = await supabase.auth.getUser()
     if (!user) throw new Error('인증이 필요합니다.')

     const [profile] = await db
       .select()
       .from(userProfiles)
       .where(eq(userProfiles.id, user.id))
       .limit(1)

     if (!profile || profile.role !== 'super_admin' || profile.deactivatedAt) {
       throw new Error('권한이 없습니다.')
     }
     return { user, profile }
   }

   export async function getMe(): Promise<{ id: string; role: 'super_admin' | 'admin'; deactivatedAt: Date | null } | null> {
     const supabase = await createClient()
     const { data: { user } } = await supabase.auth.getUser()
     if (!user) return null

     const [profile] = await db
       .select({ id: userProfiles.id, role: userProfiles.role, deactivatedAt: userProfiles.deactivatedAt })
       .from(userProfiles)
       .where(eq(userProfiles.id, user.id))
       .limit(1)

     return profile ?? null
   }
   ```

3. **Create `src/lib/admin/queries.ts`:**
   ```ts
   import 'server-only'
   import { db } from '@/lib/db'
   import { userProfiles } from '@/lib/db/schema'
   import { alias } from 'drizzle-orm/pg-core'
   import { asc, eq } from 'drizzle-orm'

   export type AccountRow = {
     id: string
     email: string
     role: 'super_admin' | 'admin'
     displayName: string | null
     createdAt: Date
     createdByEmail: string | null
     deactivatedAt: Date | null
   }

   export async function listAdmins(): Promise<AccountRow[]> {
     const creator = alias(userProfiles, 'creator')
     const rows = await db
       .select({
         id: userProfiles.id,
         email: userProfiles.email,
         role: userProfiles.role,
         displayName: userProfiles.displayName,
         createdAt: userProfiles.createdAt,
         createdByEmail: creator.email,
         deactivatedAt: userProfiles.deactivatedAt,
       })
       .from(userProfiles)
       .leftJoin(creator, eq(creator.id, userProfiles.createdBy))
       .orderBy(asc(userProfiles.createdAt))
     return rows
   }

   export async function getProfile(id: string) {
     const [row] = await db.select().from(userProfiles).where(eq(userProfiles.id, id)).limit(1)
     return row ?? null
   }
   ```

3. **Implement `src/app/(auth)/admin/accounts/actions.ts`:**
   ```ts
   'use server'
   import { revalidatePath } from 'next/cache'
   import { z } from 'zod'
   import { eq, and, isNull, sql } from 'drizzle-orm'
   import { db } from '@/lib/db'
   import { userProfiles, auditLogs, userRoleEnum, type UserRole } from '@/lib/db/schema'
   import { createAdminClient } from '@/lib/supabase/admin'
   import { assertSuperAdmin, getInitialPassword, BAN_FOREVER } from '@/lib/admin/auth-helpers'

   // ─── createAccount ──────────────────────────────────────────
   const CreateSchema = z.object({
     email: z.email({ error: '올바른 이메일을 입력하세요.' }).trim().toLowerCase(),
     role: z.enum(['super_admin', 'admin']),
     displayName: z.string().trim().max(100).optional(),
   })

   export async function createAccount(_prev: unknown, formData: FormData): Promise<{ error?: string; success?: boolean }> {
     let actor
     try { actor = await assertSuperAdmin() } catch (e) {
       return { error: e instanceof Error ? e.message : '권한 확인 실패' }
     }

     const parsed = CreateSchema.safeParse({
       email: formData.get('email'),
       role: formData.get('role'),
       displayName: (formData.get('displayName') as string) || undefined,
     })
     if (!parsed.success) return { error: parsed.error.issues[0]?.message ?? '잘못된 입력입니다.' }

     const admin = createAdminClient()
     const initial = getInitialPassword()

     const { data: created, error: createErr } = await admin.auth.admin.createUser({
       email: parsed.data.email,
       password: initial,
       email_confirm: true,
     })
     if (createErr || !created.user) {
       return { error: createErr?.message ?? '계정 생성 실패' }
     }

     try {
       await db.transaction(async (tx) => {
         await tx.insert(userProfiles).values({
           id: created.user!.id,
           email: parsed.data.email,
           role: parsed.data.role,
           displayName: parsed.data.displayName ?? null,
           createdBy: actor.user.id,
         })
         await tx.insert(auditLogs).values({
           actorId: actor.user.id,
           action: 'account.create',
           targetId: created.user!.id,
           metadata: { email: parsed.data.email, role: parsed.data.role },
         })
       })
     } catch (e) {
       await admin.auth.admin.deleteUser(created.user.id).catch(() => {})  // best-effort
       return { error: e instanceof Error ? e.message : '프로필 생성 실패' }
     }

     revalidatePath('/admin/accounts')
     return { success: true }
   }

   // ─── changeRole ─────────────────────────────────────────────
   export async function changeRole(targetId: string, newRole: UserRole): Promise<{ success: boolean; error?: string }> {
     try {
       return await db.transaction(async (tx) => {
         const supabase = await import('@/lib/supabase/server').then(m => m.createClient())
         const { data: { user } } = await supabase.auth.getUser()
         if (!user) return { success: false, error: '인증이 필요합니다.' }

         const [actorProfile] = await tx.select().from(userProfiles)
           .where(eq(userProfiles.id, user.id))
           .for('update')
         if (!actorProfile || actorProfile.role !== 'super_admin' || actorProfile.deactivatedAt) {
           return { success: false, error: '권한이 없습니다.' }
         }

         const [target] = await tx.select().from(userProfiles)
           .where(eq(userProfiles.id, targetId))
           .for('update')
         if (!target) return { success: false, error: '대상 계정을 찾을 수 없습니다.' }

         if (target.role === 'super_admin' && newRole === 'admin') {
           const [{ count }] = await tx
             .select({ count: sql<number>`count(*)::int` })
             .from(userProfiles)
             .where(and(eq(userProfiles.role, 'super_admin'), isNull(userProfiles.deactivatedAt)))
           if (count <= 1) return { success: false, error: '마지막 super_admin은 강등할 수 없습니다.' }
         }

         await tx.update(userProfiles).set({ role: newRole }).where(eq(userProfiles.id, targetId))
         await tx.insert(auditLogs).values({
           actorId: user.id,
           action: 'account.role_change',
           targetId,
           metadata: { from: target.role, to: newRole },
         })
         return { success: true }
       }).then((r) => { if (r.success) revalidatePath('/admin/accounts'); return r })
     } catch (e) {
       return { success: false, error: e instanceof Error ? e.message : '역할 변경 실패' }
     }
   }

   // ─── deactivateAccount ──────────────────────────────────────
   export async function deactivateAccount(targetId: string): Promise<{ success: boolean; error?: string }> {
     let actor
     try { actor = await assertSuperAdmin() } catch (e) { return { success: false, error: e instanceof Error ? e.message : '권한 확인 실패' } }
     if (targetId === actor.user.id) return { success: false, error: '본인 계정은 비활성화할 수 없습니다.' }

     // Last-super_admin guard inside a tx with FOR UPDATE
     try {
       const guardResult = await db.transaction(async (tx) => {
         const [target] = await tx.select().from(userProfiles)
           .where(eq(userProfiles.id, targetId)).for('update')
         if (!target) return { ok: false as const, error: '대상 계정을 찾을 수 없습니다.' }
         if (target.deactivatedAt) return { ok: false as const, error: '이미 비활성화된 계정입니다.' }
         if (target.role === 'super_admin') {
           const [{ count }] = await tx
             .select({ count: sql<number>`count(*)::int` })
             .from(userProfiles)
             .where(and(eq(userProfiles.role, 'super_admin'), isNull(userProfiles.deactivatedAt)))
           if (count <= 1) return { ok: false as const, error: '마지막 super_admin은 비활성화할 수 없습니다.' }
         }
         return { ok: true as const }
       })
       if (!guardResult.ok) return { success: false, error: guardResult.error }
     } catch (e) {
       return { success: false, error: e instanceof Error ? e.message : '비활성화 실패' }
     }

     // Apply ban_duration first (per RESEARCH ordering for Supabase + DB)
     const admin = createAdminClient()
     const { error: banErr } = await admin.auth.admin.updateUserById(targetId, { ban_duration: BAN_FOREVER })
     if (banErr) return { success: false, error: banErr.message }

     // Then mirror in DB + audit, atomically
     try {
       await db.transaction(async (tx) => {
         await tx.update(userProfiles)
           .set({ deactivatedAt: new Date(), deactivatedBy: actor.user.id })
           .where(eq(userProfiles.id, targetId))
         await tx.insert(auditLogs).values({
           actorId: actor.user.id,
           action: 'account.deactivate',
           targetId,
           metadata: { reason: 'admin_action' },
         })
       })
     } catch (e) {
       // Best-effort un-ban — DB state and Auth state diverged; log loudly
       console.warn('[deactivateAccount] DB tx failed after ban applied — attempting un-ban', { targetId, e })
       await admin.auth.admin.updateUserById(targetId, { ban_duration: 'none' }).catch(() => {})
       return { success: false, error: e instanceof Error ? e.message : '비활성화 후 DB 갱신 실패' }
     }
     revalidatePath('/admin/accounts')
     return { success: true }
   }

   // ─── reactivateAccount ──────────────────────────────────────
   export async function reactivateAccount(targetId: string): Promise<{ success: boolean; error?: string }> {
     let actor
     try { actor = await assertSuperAdmin() } catch (e) { return { success: false, error: e instanceof Error ? e.message : '권한 확인 실패' } }
     const admin = createAdminClient()
     const { error } = await admin.auth.admin.updateUserById(targetId, { ban_duration: 'none' })
     if (error) return { success: false, error: error.message }
     await db.transaction(async (tx) => {
       await tx.update(userProfiles)
         .set({ deactivatedAt: null, deactivatedBy: null })
         .where(eq(userProfiles.id, targetId))
       await tx.insert(auditLogs).values({
         actorId: actor.user.id,
         action: 'account.reactivate',
         targetId,
         metadata: {},
       })
     })
     revalidatePath('/admin/accounts')
     return { success: true }
   }
   ```
