    ```ts
    import { db } from '@/lib/db'
    import { inquiries, orders } from '@/lib/db/schema'
    import { and, eq, sql } from 'drizzle-orm'
    import type { NormalizedInquiry } from '@/lib/marketplace/types'

    export async function upsertInquiries(
      userId: string,
      marketplaceId: string,
      items: NormalizedInquiry[],
    ): Promise<{ inserted: number; updated: number }> {
      let inserted = 0
      let updated = 0
      for (const item of items) {
        // Resolve order_id (best-effort, NULL if not found)
        let orderId: string | null = null
        if (item.marketplaceOrderId) {
          const [match] = await db
            .select({ id: orders.id })
            .from(orders)
            .where(and(
              eq(orders.userId, userId),
              eq(orders.marketplaceId, marketplaceId),
              eq(orders.marketplaceOrderId, item.marketplaceOrderId),
            ))
            .limit(1)
          orderId = match?.id ?? null
        }
        const result = await db
          .insert(inquiries)
          .values({
            userId,
            marketplaceId,
            marketplaceInquiryId: item.marketplaceInquiryId,
            marketplaceOrderId: item.marketplaceOrderId ?? null,
            orderId,
            inquiryType: item.inquiryType,
            question: item.question,
            answeredAt: item.answeredAt ?? null,
            requestedAt: item.requestedAt,
            rawData: item.rawData,
          })
          .onConflictDoUpdate({
            target: [inquiries.userId, inquiries.marketplaceId, inquiries.marketplaceInquiryId],
            set: {
              answeredAt: item.answeredAt ?? null,
              rawData: item.rawData,
              updatedAt: sql`NOW()`,
            },
          })
          .returning({ id: inquiries.id, createdAt: inquiries.createdAt, updatedAt: inquiries.updatedAt })
        if (result[0] && result[0].createdAt.getTime() === result[0].updatedAt.getTime()) inserted++
        else updated++
      }
      return { inserted, updated }
    }

    export async function listInquiriesByOrderIds(orderIds: string[]) {
      if (orderIds.length === 0) return []
      const { inArray } = await import('drizzle-orm')
      return db.select().from(inquiries).where(inArray(inquiries.orderId, orderIds))
    }
    ```

    import pino from 'pino'

    import { createAdapter } from '@/lib/marketplace/registry'  // **PLACEHOLDER — replace with the import resolved in Step 1.** If the existing factory is `getAdapter` from configs.ts, use that. NO INVENTING — must match an existing export in src/lib/marketplace/.
    import { upsertInquiries } from '@/lib/orders/inquiry-queries'

    const log = pino({ name: 'inquiry-worker' })

    export const INQUIRY_QUEUE = 'inquiry-collection'

    export interface InquiryJobData {
      userId: string
      marketplaceId: string  // 'coupang' for Phase 8
      since: string  // ISO date — last collection timestamp
    }

    export const inquiryQueue = new Queue<InquiryJobData>(INQUIRY_QUEUE, { connection: redisConnection })

    export function startInquiryWorker(): Worker<InquiryJobData> {
      return new Worker<InquiryJobData>(
        INQUIRY_QUEUE,
        async (job: Job<InquiryJobData>) => {
          const { userId, marketplaceId, since } = job.data
          log.info({ userId, marketplaceId, since }, 'inquiry collection start')
          const adapter = await createAdapter(userId, marketplaceId)
          if (!adapter.getInquiries) {
            log.warn({ marketplaceId }, 'adapter does not support getInquiries — skipping')
            return { skipped: true }
          }
          const inquiries = await adapter.getInquiries(new Date(since))
          const result = await upsertInquiries(userId, marketplaceId, inquiries)
          log.info({ userId, marketplaceId, fetched: inquiries.length, ...result }, 'inquiry collection done')
          return { fetched: inquiries.length, ...result }
        },
        { connection: redisConnection, concurrency: 2 },
      )
