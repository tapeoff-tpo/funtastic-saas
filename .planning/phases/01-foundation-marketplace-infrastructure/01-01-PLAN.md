   - Import `createServerClient` from `@supabase/ssr`, `NextRequest`, `NextResponse` from `next/server`
   - Create middleware function that:
     a. Creates a Supabase server client with cookie getAll/setAll handlers (setAll must set on BOTH request and response — see research Pattern 1)
     b. Calls `const { data: { user } } = await supabase.auth.getUser()` to verify/refresh session

     c. If no user and path is not `/login` and not `/auth/callback`, redirect to `/login`
     d. If user exists and path is `/login`, redirect to `/dashboard`
     e. Set `Cache-Control: private, no-store` header on response
   - Export matcher config: `['/((?!_next/static|_next/image|favicon.ico|api/auth).*)']`

3. Create `src/app/(public)/login/actions.ts` (per D-01):
   ```typescript
   'use server'

   import { createClient } from '@/lib/supabase/server'
   import { redirect } from 'next/navigation'

   export async function login(formData: FormData) {
     const supabase = await createClient()
     const email = formData.get('email') as string
     const password = formData.get('password') as string

     if (!email || !password) {
       return { error: '이메일과 비밀번호를 입력해주세요.' }
     }

     const { error } = await supabase.auth.signInWithPassword({
       email,
       password,
     })

     if (error) {
       return { error: error.message }
     }

     redirect('/dashboard')
   }
   ```

5. Create `src/app/auth/callback/route.ts`:
   Handle Supabase auth callback (for email confirmation flows):
   ```typescript
   import { createClient } from '@/lib/supabase/server'
   import { NextResponse } from 'next/server'

   export async function GET(request: Request) {
     const { searchParams, origin } = new URL(request.url)
     const code = searchParams.get('code')
     const next = searchParams.get('next') ?? '/dashboard'

     if (code) {
       const supabase = await createClient()
       const { error } = await supabase.auth.exchangeCodeForSession(code)
       if (!error) {
         return NextResponse.redirect(`${origin}${next}`)
       }
     }

     return NextResponse.redirect(`${origin}/login?error=auth_callback_failed`)
   }
   ```
