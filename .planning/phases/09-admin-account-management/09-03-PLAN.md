Create the project's first proper `Dialog` primitive, wrapping `@base-ui/react/dialog`. This file is consumed by the AddAccountDialog and the RowActions confirm dialogs in Tasks 3–4.

**Why now:** RESEARCH says the codebase has no Dialog primitive yet (`src/components/ui/` lists badge, button, card, input, label, pagination, select, separator only). Hand-rolled dialogs in `(auth)/orders/hold-dialog.tsx` lack focus-trap, escape-key, aria-modal. Phase 9's account management dialogs are higher stakes — proper a11y matters.

**Style:** match the existing shadcn-base-nova theming used elsewhere. Reference `src/components/ui/card.tsx` for class-name conventions (rounded-lg, bg-white, shadow, p-6).

```tsx
// src/components/ui/dialog.tsx
'use client'
import { Dialog as DialogPrimitive } from '@base-ui/react/dialog'
import { cn } from '@/lib/utils'  // confirm exists; if not, use clsx/twMerge — check src/lib/utils.ts
import * as React from 'react'

export const Dialog = DialogPrimitive.Root
export const DialogTrigger = DialogPrimitive.Trigger
export const DialogPortal = DialogPrimitive.Portal
export const DialogClose = DialogPrimitive.Close

export function DialogBackdrop({ className, ...props }: React.ComponentProps<typeof DialogPrimitive.Backdrop>) {
  return (
    <DialogPrimitive.Backdrop
      {...props}
      className={cn(
        'fixed inset-0 z-40 bg-black/50 transition-opacity',
        'data-[open]:animate-in data-[open]:fade-in',
        'data-[closed]:animate-out data-[closed]:fade-out',
        className,
      )}
    />
  )
}

export function DialogPopup({ className, ...props }: React.ComponentProps<typeof DialogPrimitive.Popup>) {
  return (
    <DialogPrimitive.Popup
      {...props}
      className={cn(
        'fixed left-1/2 top-1/2 z-50 w-full max-w-md -translate-x-1/2 -translate-y-1/2',
        'rounded-lg bg-white p-6 shadow-xl outline-none',
        'data-[open]:animate-in data-[open]:zoom-in-95',
        'data-[closed]:animate-out data-[closed]:zoom-out-95',
        className,
      )}
    />
  )
}

export const DialogTitle = ({ className, ...props }: React.ComponentProps<typeof DialogPrimitive.Title>) => (
  <DialogPrimitive.Title {...props} className={cn('text-lg font-semibold', className)} />
)

export const DialogDescription = ({ className, ...props }: React.ComponentProps<typeof DialogPrimitive.Description>) => (
  <DialogPrimitive.Description {...props} className={cn('text-sm text-muted-foreground mt-1', className)} />
)
```

**Note on `cn`:** verify `src/lib/utils.ts` exports `cn` (standard shadcn utility). If absent, either add it (`export const cn = (...args) => clsx(args)` using `clsx` + `tailwind-merge` if both installed) or inline the className concat. Confirm by `ls src/lib/utils.ts` first.

**Note on `onOpenChange`:** consumers of `Dialog.Root` must use the @base-ui signature `(open, eventDetails) => void`, not the Radix-style `(open) => void`. Document this in a JSDoc comment above the export of `Dialog`.

Client component. Trigger = a "새 관리자 추가" button. On click opens Dialog containing form bound to `createAccount` via `useActionState`. On success: dialog closes, sonner toast. The form fields are: email (required), role (Select: 관리자/오너 — defaults to 관리자), displayName (optional).

```tsx
'use client'
import { useActionState, useEffect, useState } from 'react'
import { toast } from 'sonner'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'
import { Label } from '@/components/ui/label'
import { Dialog, DialogTrigger, DialogPortal, DialogBackdrop, DialogPopup, DialogTitle, DialogDescription, DialogClose } from '@/components/ui/dialog'
import { createAccount } from './actions'

const initialState: { error?: string; success?: boolean } = {}

export function AddAccountDialog() {
  const [open, setOpen] = useState(false)
  const [state, formAction, pending] = useActionState(createAccount, initialState)

  useEffect(() => {
    if (state.success) {
      toast.success('새 관리자 계정이 생성되었습니다.')
      setOpen(false)
    } else if (state.error) {
      toast.error(state.error)
    }
  }, [state])

  return (
    <Dialog open={open} onOpenChange={(nextOpen /* note: @base-ui passes (open, eventDetails) */) => setOpen(nextOpen)}>
      <DialogTrigger render={<Button>새 관리자 추가</Button>} />
      <DialogPortal>
        <DialogBackdrop />
        <DialogPopup>
          <DialogTitle>새 관리자 추가</DialogTitle>
          <DialogDescription>이메일과 역할을 입력하세요. 초기 비밀번호는 자동 적용됩니다.</DialogDescription>
          <form action={formAction} className="mt-4 space-y-3">
            <div>
              <Label htmlFor="email">이메일</Label>
              <Input id="email" name="email" type="email" required />
            </div>
            <div>
              <Label htmlFor="role">역할</Label>
              <select id="role" name="role" required defaultValue="admin" className="block w-full rounded border px-3 py-2 text-sm">
                <option value="admin">관리자</option>
                <option value="super_admin">오너 (super_admin)</option>
              </select>
            </div>
            <div>
              <Label htmlFor="displayName">표시명 (선택)</Label>
              <Input id="displayName" name="displayName" type="text" maxLength={100} />
            </div>
            {state.error && <p className="text-sm text-red-600">{state.error}</p>}
            <div className="flex justify-end gap-2 pt-2">
              <DialogClose render={<Button type="button" variant="outline">취소</Button>} />
              <Button type="submit" disabled={pending}>{pending ? '처리 중...' : '생성'}</Button>
            </div>
          </form>
        </DialogPopup>
      </DialogPortal>
    </Dialog>
  )
}
```

**Note on `DialogTrigger render={<Button>...}`:** @base-ui supports a `render` prop for trigger to compose with custom components. Verify in `node_modules/@base-ui/react/dialog/trigger/DialogTrigger.d.ts` if the prop name is `render` or `children` with `asChild` semantics. If different, adapt — the goal is "Button is the trigger." If unclear, fall back to plain `<DialogTrigger><Button>...</Button></DialogTrigger>`.

**Decision (per D-discretion — Select):** uses native `<select>` for role since the project's `select.tsx` is a styled `<select>` wrapper (confirm). If a styled component exists with matching API, use it. Native select keeps form binding simple.

Client component rendering per-row action controls. Three confirm-dialogs + one role-change dialog. Each driven by `useTransition` + sonner toast.

Approach — to avoid one giant component, structure as 4 small inline action components that each render a Dialog + trigger button. The `RowActions` parent renders them inline. (Alternative: use a dropdown menu — but the project has no menu primitive yet, and adding one is out of scope. Inline buttons are acceptable for ≤20 rows.)

```tsx
'use client'
import { useState, useTransition } from 'react'
import { toast } from 'sonner'
import { Button } from '@/components/ui/button'
import { Dialog, DialogTrigger, DialogPortal, DialogBackdrop, DialogPopup, DialogTitle, DialogDescription, DialogClose } from '@/components/ui/dialog'
import type { AccountRow } from '@/lib/admin/queries'
import { changeRole, resetAccountPassword, deactivateAccount, reactivateAccount } from './actions'

export function RowActions({ account }: { account: AccountRow }) {
  return (
    <div className="flex flex-wrap gap-2">
      <ChangeRoleAction account={account} />
      <ResetPasswordAction account={account} />
      {account.deactivatedAt
        ? <ReactivateAction account={account} />
        : <DeactivateAction account={account} />
      }
    </div>
  )
}

function ChangeRoleAction({ account }: { account: AccountRow }) {
  const [open, setOpen] = useState(false)
  const [pending, startTransition] = useTransition()
  const next: 'super_admin' | 'admin' = account.role === 'super_admin' ? 'admin' : 'super_admin'
  const nextLabel = next === 'super_admin' ? '오너로 승격' : '관리자로 강등'
  return (
    <Dialog open={open} onOpenChange={(o) => setOpen(o)}>
      <DialogTrigger render={<Button size="sm" variant="outline">역할 변경</Button>} />
      <DialogPortal>
        <DialogBackdrop />
        <DialogPopup>
          <DialogTitle>역할 변경</DialogTitle>
          <DialogDescription>{account.email}을 {nextLabel}하시겠습니까?</DialogDescription>
          <div className="mt-4 flex justify-end gap-2">
            <DialogClose render={<Button variant="outline">취소</Button>} />
            <Button
              disabled={pending}
              onClick={() => startTransition(async () => {
                const r = await changeRole(account.id, next)
                if (r.success) { toast.success('역할이 변경되었습니다.'); setOpen(false) }
                else toast.error(r.error ?? '역할 변경 실패')
              })}
            >{pending ? '처리 중...' : '변경'}</Button>
          </div>
        </DialogPopup>
      </DialogPortal>
    </Dialog>
  )
}

function ResetPasswordAction({ account }: { account: AccountRow }) {
  const [open, setOpen] = useState(false)
  const [pending, startTransition] = useTransition()
  return (
    <Dialog open={open} onOpenChange={(o) => setOpen(o)}>

      <DialogPortal>
        <DialogBackdrop />
        <DialogPopup>
          <DialogTitle>비밀번호 초기화</DialogTitle>

          <div className="mt-4 flex justify-end gap-2">
            <DialogClose render={<Button variant="outline">취소</Button>} />
            <Button
              disabled={pending}
              onClick={() => startTransition(async () => {
                const r = await resetAccountPassword(account.id)
                if (r.success) { toast.success('비밀번호가 초기화되었습니다.'); setOpen(false) }
                else toast.error(r.error ?? '초기화 실패')
              })}
            >{pending ? '처리 중...' : '초기화'}</Button>
          </div>
        </DialogPopup>
      </DialogPortal>
    </Dialog>
  )
}

function DeactivateAction({ account }: { account: AccountRow }) {
  const [open, setOpen] = useState(false)
  const [pending, startTransition] = useTransition()
  return (
    <Dialog open={open} onOpenChange={(o) => setOpen(o)}>
      <DialogTrigger render={<Button size="sm" variant="destructive">비활성화</Button>} />
      <DialogPortal>
        <DialogBackdrop />
        <DialogPopup>
          <DialogTitle>계정 비활성화</DialogTitle>
          <DialogDescription>{account.email} 계정을 비활성화합니다. 로그인이 차단됩니다.</DialogDescription>

            <DialogClose render={<Button variant="outline">취소</Button>} />
            <Button
              variant="destructive"
              disabled={pending}

                const r = await deactivateAccount(account.id)
                if (r.success) { toast.success('계정이 비활성화되었습니다.'); setOpen(false) }
                else toast.error(r.error ?? '비활성화 실패')
              })}
            >{pending ? '처리 중...' : '비활성화'}</Button>
          </div>
        </DialogPopup>
      </DialogPortal>
    </Dialog>
  )
}

function ReactivateAction({ account }: { account: AccountRow }) {
  const [open, setOpen] = useState(false)
  const [pending, startTransition] = useTransition()
  return (
    <Dialog open={open} onOpenChange={(o) => setOpen(o)}>
      <DialogTrigger render={<Button size="sm" variant="outline">재활성화</Button>} />

        <DialogBackdrop />
        <DialogPopup>
          <DialogTitle>계정 재활성화</DialogTitle>
          <DialogDescription>{account.email} 계정을 다시 활성화합니다.</DialogDescription>
          <div className="mt-4 flex justify-end gap-2">
            <DialogClose render={<Button variant="outline">취소</Button>} />

              })}
            >{pending ? '처리 중...' : '재활성화'}</Button>
          </div>
        </DialogPopup>
      </DialogPortal>
    </Dialog>
  )
}
```

2. **`src/app/(auth)/settings/account/password-form.tsx`** (client form):
   ```tsx
   'use client'
   import { useActionState, useEffect, useRef } from 'react'
   import { toast } from 'sonner'
   import { Button } from '@/components/ui/button'
   import { Input } from '@/components/ui/input'
   import { Label } from '@/components/ui/label'
   import { changeOwnPassword } from './password-actions'

   const initialState: { error?: string; success?: boolean } = {}

   export function PasswordForm() {
     const [state, formAction, pending] = useActionState(changeOwnPassword, initialState)
     const formRef = useRef<HTMLFormElement>(null)

     useEffect(() => {
       if (state.success) { toast.success('비밀번호가 변경되었습니다.'); formRef.current?.reset() }
       else if (state.error) toast.error(state.error)
     }, [state])

     return (
       <form ref={formRef} action={formAction} className="space-y-3">
         <div>
           <Label htmlFor="newPassword">새 비밀번호</Label>
           <Input id="newPassword" name="newPassword" type="password" required minLength={8} maxLength={72} />
         </div>
         {state.error && <p className="text-sm text-red-600">{state.error}</p>}
         <Button type="submit" disabled={pending}>{pending ? '처리 중...' : '변경'}</Button>
       </form>
     )
   }
   ```

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-17 | Information Disclosure | Server-side `getMe()` result rendered into client bundle leaks PII | accept | `getMe()` returns only `{ id, role, deactivatedAt }` — no PII. Account page renders email/displayName, but those are owner-visible by design. |
